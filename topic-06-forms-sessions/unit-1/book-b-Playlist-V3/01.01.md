  async removePlaylist(id) {
    const query = 'DELETE FROM playlist_playlists WHERE id=$1';
    const values = [id];
    try {
      await dataStoreClient.query(query, values);
    } catch (e) {
      logger.error("Unable to remove playlist:", e);
    }
  },
